COMMENT ON COLUMN "REND_ADJUNTOS"."ADJ_RUTA" IS 'Ruta relativa del archivo';

COMMENT ON COLUMN "REND_ADJUNTOS"."ADJ_TAMANO" IS 'Tamaño en bytes';
COMMENT ON COLUMN "REND_ADJUNTOS"."ADJ_DESCRIPCION" IS 'Descripción opcional';
